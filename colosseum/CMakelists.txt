cmake_minimum_required(VERSION 3.16)
project(Colosseum LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20)

add_library(cli-lib STATIC
    src/cli/parser.cpp
)

target_include_directories(cli-lib PRIVATE
    src/cli
)

add_executable(colosseum-cli 
    colosseum-cli.cc
)

target_include_directories(colosseum-cli PRIVATE
    src
    src/cli
)

target_link_libraries(colosseum-cli PRIVATE cli-lib)
